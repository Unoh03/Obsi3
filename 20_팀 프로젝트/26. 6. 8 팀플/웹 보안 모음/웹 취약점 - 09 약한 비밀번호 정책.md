# 웹 취약점 - 09 약한 비밀번호 정책
```php
function validate_password_policy($pw, $id = '') {
    $commonPasswords = [
        '1234', '12345', '123456', '12345678',
        '1111', '0000', 'password', 'qwer1234',
        'admin', 'admin1234'
    ];

    if (strlen($pw) < 8) {
        return '비밀번호는 8자 이상이어야 합니다.';
    }

    if (in_array(strtolower($pw), $commonPasswords, true)) {
        return '너무 흔한 비밀번호는 사용할 수 없습니다.';
    }

    if ($id !== '' && stripos($pw, $id) !== false) {
        return '비밀번호에 아이디를 포함할 수 없습니다.';
    }

    if (!preg_match('/[A-Z]/', $pw)) {
        return '비밀번호에는 영문 대문자가 1자 이상 포함되어야 합니다.';
    }

    if (!preg_match('/[a-z]/', $pw)) {
        return '비밀번호에는 영문 소문자가 1자 이상 포함되어야 합니다.';
    }

    if (!preg_match('/[0-9]/', $pw)) {
        return '비밀번호에는 숫자가 1자 이상 포함되어야 합니다.';
    }

    if (!preg_match('/[^A-Za-z0-9]/', $pw)) {
        return '비밀번호에는 특수문자가 1자 이상 포함되어야 합니다.';
    }

    if (preg_match('/(.)\1\1/', $pw)) {
        return '같은 문자를 3번 이상 연속으로 사용할 수 없습니다.';
    }

    return '';
}

$passwordError = validate_password_policy($pw, $id);

if ($passwordError !== '') {
    echo "<script>alert('" . addslashes($passwordError) . "'); history.go(-1);</script>";
    exit;
}
```
