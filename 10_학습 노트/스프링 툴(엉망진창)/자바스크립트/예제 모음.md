```HTML
<%@ page language="java" contentType="text/html; charset=UTF-8"
	pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>ex04</title>
</head>
<body>
	<script>
/* 	
 	true && true  -> true
	
	true && false -> false
	false && true -> false
	
	false && false -> false
	------------------------------
	true || true  -> true
	
	true || false -> true
	false || true -> true
	
	false && false -> false 
*/
	
	var x = 5, y = 7;
	document.write("(x < 10 && y > 10) : " + (x < 10 && y > 10) + "<br>");
	document.write("(x < 10 || y > 10) : " + (x < 10 || y > 10) + "<br>");
	document.write("!(x < 10 && y > 10) : " + !(x < 10 && y > 10) + "<br>");

	// true or false ? true일때 실행할 문장 : false일 때 실행할 문장;
		
	result = (x > y) ? x : y; // 조건 연산
	document.write("큰 값 : " + result + "<br>");
	result = (x < y) ? x - y : y - x; // 조건 연산
	document.write("큰 값 - 작은 값 : " + result + "<br>");
	</script>
</body>
```

```html
		if (pwValue.length <= 0) {

			alert("비번 5자리 입력.");

	function realTimepwCheck() {
    // 🚨 주의: name 속성으로 바로 접근(pw.value)하는 건 구식(Legacy) 브라우저의 잔재다.
    // 실무에서는 무조건 document.getElementsByName('pw')[0].value 를 쓰거나,
    // input 태그에 id="pw"를 주고 document.getElementById('pw').value 를 써야 안전하다.
    let pwValue = document.getElementsByName('pw')[0].value;
    
    
    if (pwValue.length <= 0) {
        document.getElementById('pwmsg').innerHTML = "비번 입력";
    } else if (pwValue.length < 4) {
        document.getElementById('pwmsg').innerHTML = "비번 5자리 이상 입력";
    } else {
        // 4자리 이상 정상 입력됐을 때 경고창을 지워주는 로직도 필요하다!

	}
}
```
