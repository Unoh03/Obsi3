[R1]
conf t

! 인터페 추가
int f0/0
ip add 192.168.1.254 255.255.255.0
no sh
int f0/1
ip add 12.12.12.1 255.255.255.0
no sh
exit

! 라우팅
router r
version 2
no au
net 12.12.12.0
!192.168.1.0은 보안 상의 이유로 광고 안함
exit

! NAT (PAT - Overload: 내부 PC들이 12.12.12.1 공인 IP 하나를 공유하여 나감)
ac 1 p 192.168.1.0 0.0.0.255
ip nat inside source list 1 int f0/1 overload
int f0/0
ip nat inside
int f0/1
ip nat outside
end
cop r s
