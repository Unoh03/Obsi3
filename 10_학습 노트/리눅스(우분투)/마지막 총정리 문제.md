## **[[dns.sh]]**
```bash
#!/bin/bash
# ==========================================
#[DNS Server Provisioning Script]
# 역할: BIND9 설치 및 Zone File 구성
# ==========================================
sudo apt update
sudo apt install -y bind9 bind9utils bind9-doc

# 1. 글로벌 옵션 설정 (보안을 위해 recursion은 no로 설정하여 Open Resolver 방지)
sudo tee /etc/bind/named.conf.options > /dev/null <<EOF
options {
    directory "/var/cache/bind";
    recursion no; # example.com 만 접속할거니깐 no. 다른 곳도 들어가려면 yes
    allow-query { any; };
    listen-on-v6 { any; };
};
EOF

# 2. 도메인 Zone 선언
sudo tee /etc/bind/named.conf.local > /dev/null <<EOF
zone "example.com" {
    type master;
    file "/etc/bind/web.example.com";
};
EOF

# 3. Zone File (호적 등본) 생성
# 주의: $TTL 앞의 역슬래시(\)는 쉘이 변수로 착각하는 것을 막기 위한 이스케이프 처리임.
sudo tee /etc/bind/web.example.com > /dev/null <<EOF
\$TTL    604800
@       IN      SOA     ns.example.com. root.example.com. (
                              2         ; Serial (수정 시마다 +1)
                         604800         ; Refresh
                          86400         ; Retry
                        2419200         ; Expire
                         604800 )       ; Negative Cache TTL
;
@       IN      NS      ns.example.com.
ns      IN      A       <DNS 공인 IP>
@       IN      A       <WEB 공인 IP>
www     IN      A       <WEB 공인 IP>
EOF

# 4. 🚨 DNS 서비스 포트(53) 방화벽 개방
sudo ufw allow 53
# 5. 문법 검증 및 데몬 재시작 (메모리 적재)
sudo named-checkconf
sudo named-checkzone example.com /etc/bind/web.example.com
sudo systemctl restart named
sudo systemctl enable named
```

## **[[web.sh]]**
>[! warning] 자바, 톰캣 버젼 확인! 특히 톰캣은 버젼 업뎃하면 링크가 없어짐!
```bash
#!/bin/bash
# ==========================================
#[WEB Server Provisioning Script]
# 역할: Tomcat 10 설치, 보안 계정 격리, WAR 자동 배포
# ==========================================
echo "[INFO] Tomcat 웹 서버 세팅을 시작합니다..."

sudo apt update
sudo apt install -y openjdk-17-jdk
wget http://mirror.apache-kr.org/apache/tomcat/tomcat-10/v10.1.53/bin/apache-tomcat-10.1.53.tar.gz

# 🛡️ [보안] 쉘 접속이 불가능한(/bin/false) 톰캣 전용 계정 생성 및 권한 격리
sudo useradd -r -m -U -d /opt/tomcat -s /bin/false tomcat
sudo tar -xf apache-tomcat-10.1.53.tar.gz -C /opt/tomcat
sudo mv /opt/tomcat/apache-tomcat-10.1.53 /opt/tomcat/tomcat-10
sudo chown -RH tomcat: /opt/tomcat/tomcat-10

# 1. tomcat.service 파일을 스크립트 내부에서 직접 생성 (tee 명령어 활용)

sudo tee /etc/systemd/system/tomcat.service > /dev/null <<EOF
[Unit]
Description=Tomcat 10 servlet container

After=network.target

  

[Service]
Type=forking
User=tomcat
Group=tomcat
Environment="JAVA_HOME=/usr/lib/jvm/java-17-openjdk-amd64/"
ExecStart=/opt/tomcat/tomcat-10/bin/startup.sh
ExecStop=/opt/tomcat/tomcat-10/bin/shutdown.sh

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable --now tomcat

# 2. war 파일 이동 (boot.war 파일이 스크립트와 같은 폴더에 있다고 가정)
if [ -f "boot.war" ]; then
    sudo mv boot.war /opt/tomcat/tomcat-10/webapps/
        # 🚨[보안 패치] root로 옮긴 파일의 소유권을 tomcat으로 변경하여 Permission Denied 방지
    sudo chown tomcat:tomcat /opt/tomcat/tomcat-10/webapps/boot.war
    echo "[INFO] boot.war 배포 완료. Tomcat이 압축을 해제할 때까지 대기합니다..."
else
    echo "[ERROR] boot.war 파일이 없습니다! 스크립트를 중단합니다."
    exit 1
fi

# 3. Smart Polling: 파일이 생성될 때까지 대기 (최대 30초)
PROP_FILE="/opt/tomcat/tomcat-10/webapps/boot/WEB-INF/classes/application.properties"
WAIT_TIME=0

# sudo test -f 로 권한 문제 우회! 파일이 없을(! ) 동안 계속 루프를 돈다.
while ! sudo test -f "$PROP_FILE"; do
    sleep 1
    WAIT_TIME=$((WAIT_TIME + 1))
    echo "압축 해제 대기 중... (${WAIT_TIME}초 경과)"
    
    if [ $WAIT_TIME -ge 30 ]; then
        echo "[ERROR] 30초가 지났지만 파일이 생성되지 않았습니다. 수동 확인 요망!"
        exit 1
    fi
done

# 4. application.properties 자동 수정
echo "[SUCCESS] application.properties 파일 발견! 설정을 변경합니다."
sudo sed -i 's|spring.datasource.username.*|spring.datasource.username=<DB_ID>|' $PROP_FILE
sudo sed -i 's|spring.datasource.password.*|spring.datasource.password=<DB_PW>|' $PROP_FILE
sudo sed -i 's|spring.datasource.url.*|spring.datasource.url=jdbc:mariadb://<DB 서버 공인 IP>:3306/care|' $PROP_FILE

# 5. 🚨 [보안 패치] WEB 서비스 포트(8080) 방화벽 개방
sudo ufw allow 8080/tcp
# 6. 재시작
sudo systemctl restart tomcat
echo "[SUCCESS] WEB 서버 세팅 및 DB 연동이 완벽하게 끝났습니다!"
```

```bash
#!/bin/bash
echo "[INFO] Load-Balancer 설치 및 세팅을 시작합니다..."
sudo apt update
sudo apt install nginx -y

sudo tee /etc/nginx/conf.d/load-balancer.conf > /dev/null << 'EOF'

upstream backend_nodes {
    server 192.168.3.1:8080;
    server 192.168.3.2:8080;
    ip_hash;
}
server {
    listen 80;
    server_name _;

    location / {
        proxy_pass http://backend_nodes;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
EOF

sudo rm -rf /etc/nginx/sites-enabled/default

# Nginx 문법 검사 (오타 방지)
sudo nginx -t
# Nginx 설정 적용 및 재시작
sudo systemctl restart nginx
```
```nginx
# ============================================
# [upstream 블록]: 로드밸런싱 대상 서버 그룹 정의
# "backend_nodes"라는 이름으로 묶어서, 아래 server 블록에서 이름으로 호출함
# ============================================
upstream backend_nodes {
    # 기본 알고리즘: Round Robin (위→아래 순서로 번갈아 분배)
    # 별도 선언 없으면 자동 적용됨
    server 192.168.3.1:8080;  # WEB1 (사설 IP:포트)
    server 192.168.3.2:8080;  # WEB2 (사설 IP:포트)
}

# ============================================
# [server 블록]: 클라이언트의 요청을 받는 LB 자신의 설정
# ============================================
server {
    listen 80;       # LB가 80포트(HTTP)로 클라이언트 요청을 대기
    server_name _;   # "_" = 도메인 상관없이 들어오는 모든 요청 수락

    location / {     # "/" = 모든 경로의 요청에 대해 아래 설정 적용

        # 들어온 요청을 upstream 그룹(backend_nodes)으로 넘김
        # 실제로는 WEB1 또는 WEB2로 번갈아 포워딩됨
        proxy_pass http://backend_nodes;

        # LB를 거치면 WEB 서버 입장에선 요청자가 LB로 보임
        # 아래 헤더들이 없으면 WEB 서버 로그에 클라이언트 IP 대신 LB IP만 찍힘
        proxy_set_header Host $host;
        # 클라이언트가 요청한 원래 도메인명 전달 (예: www.example.com)

        proxy_set_header X-Real-IP $remote_addr;
        # 클라이언트의 진짜 IP 전달 (WEB 서버 로그용)

        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        # 프록시 경유 이력 전달. 체인 형태로 누적됨
        # 예: "클라이언트IP, 프록시1IP, 프록시2IP"

        proxy_set_header X-Forwarded-Proto $scheme;
        # 클라이언트가 HTTP/HTTPS 중 뭘로 왔는지 전달
        # WEB 서버가 리다이렉트 처리할 때 필요
    }
}
```

```bash
# Nginx 설정 적용 및 재시작

```
>[! warning] - **팩트 폭격:** Nginx를 처음 설치하면 저 `default` 파일 안에 **`listen 80 default_server;`** 라는 설정이 하드코딩되어 있다.
- **대참사 시나리오:** 저 파일을 안 지우고 강도님의 `load-balancer.conf` (여기도 `listen 80;`이 있음)를 넣은 뒤 Nginx를 재시작하면? 
- **결과:** 커널이 **"야! 80번 포트를 두 명(default, load-balancer)이 동시에 쓰겠다고 싸우잖아!"** 라며 `Address already in use` 또는 `Duplicate listen options` 에러를 뱉고 Nginx 데몬을 그 자리에서 즉사시킨다. 
- **결론:** 80번 포트의 독점권(Binding)을 가져오기 위해 기존 세입자(`default`)를 물리적으로 철거(rm)한 완벽한 조치다.
- 실습: 그런거 없고 쟤가 걍 우선권 가져서 웰컴 띄움
### 🚨 2. 클로드가 빼먹은 L7 통신의 반쪽: Tomcat `RemoteIpValve`
> **클로드의 주석:** `# 아래 헤더들이 없으면 WEB 서버 로그에 클라이언트 IP 대신 LB IP만 찍힘`

- **팩트 폭격:** 클로드의 말대로 Nginx가 `X-Forwarded-For` 헤더에 진짜 PC의 IP(`192.168.1.10`)를 담아서 톰캣으로 던져주는 건 맞다. **하지만 톰캣은 기본적으로 저 헤더를 읽을 지능이 없다.**
- **대참사 시나리오:** Nginx가 아무리 진짜 IP를 던져줘도, 톰캣은 그냥 자기한테 직접 패킷을 보낸 놈(LB의 IP인 `192.168.3.3`)만 로그(`catalina.out`)에 주구장창 찍어댄다. 나중에 해킹 사고 터져서 로그 까보면 전 세계 모든 해커의 IP가 `192.168.3.3`으로 찍혀있는 기적을 보게 된다.
- **✅ 수석 아키텍트의 완벽한 패치 (Tomcat 설정 추가):**
  Nginx가 던져준 헤더를 톰캣이 해독하게 만들려면, **WEB 서버(Tomcat)의 `/opt/tomcat/tomcat-10/conf/server.xml` 파일**을 열고 `<Host>` 태그 안에 아래의 **'밸브(Valve)'**를 반드시 끼워 넣어야 한다.

```xml
<!-- Tomcat이 Nginx가 보낸 X-Forwarded-For 헤더를 까서 진짜 IP를 로그에 기록하게 만드는 마법의 밸브 -->
<Valve className="org.apache.catalina.valves.RemoteIpValve"
       remoteIpHeader="x-forwarded-for"
       protocolHeader="x-forwarded-proto" />
```
